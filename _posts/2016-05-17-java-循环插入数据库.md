### mysql

```
DROP PROCEDURE
IF EXISTS test_insert;
DELIMITER ;;
CREATE PROCEDURE test_insert ()
BEGIN
    DECLARE
        i INT DEFAULT 1 ;
    WHILE i < 1000 DO
        INSERT INTO student (NAME, age)
    VALUES
        (“ssss”, 30) ;
    SET i = i + 1 ;
    END
    WHILE ;
    END;;
CALL test_insert () ;
```
